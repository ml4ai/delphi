   void Config::read_config() {
     ifstream file;
     file.open(filename);
     string jsonString;
     string line;
     if ( file.is_open() ) {
       while ( file ) { 
         getline (file, line);
         jsonString += line;
       }
       file.close();
       config = json::parse(jsonString);
     }
     else {
       cerr << "Couldn't open config file at '" << filename << "'" << endl;
     }
   }
